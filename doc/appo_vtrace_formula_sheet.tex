\section*{1. Return / TD}
\[
G_t = r_t + \gamma r_{t+1} + \gamma^2 r_{t+2} + \cdots
\]
\[
\delta_t = r_t + \gamma V(s_{t+1}) - V(s_t)
\]

\section*{2. Policy Gradient}
\[
J(\theta) = \mathbb{E}_{\tau\sim\pi_\theta}\left[\sum_t \gamma^t r_t\right]
\]
\[
\nabla_\theta J(\theta)=
\mathbb{E}\left[\nabla_\theta\log\pi_\theta(a_t|s_t)\,Q^\pi(s_t,a_t)\right]
\]
\[
\nabla_\theta J(\theta)=
\mathbb{E}\left[\nabla_\theta\log\pi_\theta(a_t|s_t)\,A_t\right],\quad
A_t=Q_t-V_t
\]

\section*{3. Actor-Critic Loss}
\[
L_v=\mathbb{E}\left[(V_\theta(s_t)-V_t^{\text{target}})^2\right]
\]
\[
L_{pg}=-\mathbb{E}\left[\log\pi_\theta(a_t|s_t)\,A_t\right]
\]
\[
L_{ent}=-\mathbb{E}\left[H(\pi_\theta(\cdot|s_t))\right]
\]
\[
L=L_{pg}+c_v L_v+c_{ent}L_{ent}
\]

\section*{4. PPO Clip}
\[
r_t(\theta)=\frac{\pi_\theta(a_t|s_t)}{\pi_{\text{old}}(a_t|s_t)}
\]
\[
L_{\text{clip}}=
\mathbb{E}\left[
\min\left(r_tA_t,\;\text{clip}(r_t,1-\epsilon,1+\epsilon)A_t\right)
\right]
\]

\section*{5. V-trace}
\[
\rho_t=\min\left(\bar{\rho},\frac{\pi(a_t|s_t)}{\mu(a_t|s_t)}\right),\quad
c_t=\min\left(\bar{c},\frac{\pi(a_t|s_t)}{\mu(a_t|s_t)}\right)
\]
\[
\delta_t^V=
\rho_t\left(r_t+\gamma V(s_{t+1})-V(s_t)\right)
\]
\[
v_s=
V(s)+
\sum_{t=s}^{s+n-1}
\gamma^{t-s}
\left(\prod_{i=s}^{t-1}c_i\right)\delta_t^V
\]
\[
A_t\approx \rho_t\left(r_t+\gamma v_{t+1}-V(s_t)\right)
\]
